JavaScript - 배열

1.배열의 선언

var myArray = [];

const fruits = ['apple','banana','melon'];

2.배열 불러오기

console.log(fruits);

(3) ["apple", "banana", "melon"]

3.배열의 길이 구하기 - length

console.log(fruits.length);

    3

4.인덱스 값으로 배열 출력

console.log(fruits[0]);    apple

console.log(fruits[fruits.length - 1]);

    melon

5.배열의 인자들을 하나씩 출력

1. for
for(let i = 0; i < fruits.length; i++){
console.log(fruits[i]);
}

apple banana melon

2.for of
for(let fruit of fruits){
console.log(fruit);
}

apple banana melon

6.함수를 이용한 배열의 정보

fruits.forEach(function(fruit,index,array){
console.log(fruit,index,array);
});

apple 0 (3)["apple", "banana", "melon"] banana 1 (3)"apple", "banana", "melon"] melon 2 (3) ["apple", "banana", "melon"]

7.배열의 추가와 삭제

1) push: 끝에 아이템 추가

fruits.push('new lemon','new kiwi');

console.log(fruits);

(5) ["apple", "banana", "melon", "new lemon", "new kiwi"]

2) pop: 끝에 아이템 삭제

fruits.pop();

console.log(fruits);

(4) ["apple", "banana", "melon", "new lemon"]

3) unshift: 앞에 아이템 추가

fruits.unshift('watermelon');

(5) ["watermelon", "apple", "banana", "melon", "new lemon"]

4) shift: 가장 앞에 아이템 삭제

fruits.shift();

(4) ["apple", "banana", "melon", "new lemon"]

5) splice: 인덱스 값을 통해 삭제

array.splice(시작점,삭제 할 갯수,대체품);

fruits.splice(2);

만약 삭제 할 갯수를 정하지 않으면 시작점부터 끝까지 모두 삭제

(2) ["apple", "banana"]

fruits.splice(1,1,'new fruits1','new fruits2');

(3) ["apple", "new fruits1", "new fruits2"]

8.배열 합치기 - concat

const animals = ['cat','dog','rabbit'];

const newarray = fruits.concat(animals);

console.log(newarray);

(6) ["apple", "new fruits1", "new fruits2", "cat", "dog", "rabbit"]

9.배열에서의 검색

1) 인덱스 값: index of

console.log(fruits.indexOf('apple'));

0

*없는 개체를 찾는경우 -1 을 반환


2) 배열에서 포함하고 있는지 아닌지 - true/false 변환

console.log(fruits.includes('apple'));

true
